// Button zum Laden von Posts in der NÃ¤he hinzufÃ¼gen
private addLoadPostsButton(): void {
  if (!this.map) return;
  
  // Erstelle einen benutzerdefinierten Steuerungsbutton
  const loadPostsControl = L.Control.extend({
    options: {
      position: 'bottomright'
    },
    
    onAdd: () => {
      const container = L.DomUtil.create('div', 'leaflet-bar leaflet-control posts-button');
      container.innerHTML = '<a href="#" title="Posts in der Umgebung laden"><span class="posts-icon">ðŸ“‹</span></a>';
      
      container.onclick = () => {
        if (this.map) {
          const center = this.map.getCenter();
          // Lade Posts mit Koordinaten in der aktuellen Kartenansicht
          this.loadPostsWithCoordinates(center.lng, center.lat);
        }
        return false;
      };
      
      return container;
    }
  });
  
  // Button zur Karte hinzufÃ¼gen
  this.map.addControl(new loadPostsControl());
}
